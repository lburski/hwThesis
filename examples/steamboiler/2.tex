Auxiliary Functions

\draschema{A5}{
\begin{axdef}
pswitch: (OnOff \times WorksBroken) \fun OnOff
\where
pswitch(on, works) = on \\
pswitch(on, broken) = off \\ 
pswitch(off, works) = off \\
pswitch(off,broken) = off
\end{axdef}}

\draschema{A6}{
\begin{axdef}
pamount: (OnOff \times WorksBroken) \fun \nat
\where
\forall x:OnOff, y: WorksBroken | x = off \lor y = broken @ \\
pamount(x, y) = 0 \\
pamount(on, works) = 1
\end{axdef}}
